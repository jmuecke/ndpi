Value Store

ndpi.value_store.ValueStore is a paper-agnostic store for named scalar values, emitted via one of two output strategies: a values.tex file (for \getval in the paper) or a Markdown table (for presentations/notes where values are copied by hand). Metric definitions (the polars queries) live in the paper repository; this module owns collection, validation, and emission.

Intended workflow:

import polars as pl
from ndpi.value_store import ValueStore, fmt_eng, fmt_percent

store = ValueStore()

# 1. register 1x1 LazyFrame queries (shared load_*() constructors benefit
#    from common-subplan elimination in the batched collect)
lf = load_scans()
store.register("num-scanners", lf.select(pl.col("scanner").n_unique()), fmt=fmt_eng())
store.register("_failed", lf.filter(pl.col("failed")).select(pl.len()), emit=False)
store.register("_total", lf.select(pl.len()), emit=False)

# hardcoded values are allowed but should carry a loud note
store.register("cloudflare-ech-domains", 2_100_000, fmt=fmt_eng(),
               note="FIXME hardcoded, see notebook 07")

# 2. collect: batched by default, individual=True for bounded memory
store.collect()

# 3. derived values are plain Python between collect() and write()
store.register("failed-share", 100 * store["_failed"] / store["_total"],
               fmt=fmt_percent(digits=1))

# 4. emit — pick one or both output strategies
store.write("values.tex")
store.write_markdown("values.md")

In the paper, \input{values.tex} and use values via \getval:

We observed \getval{num-scanners} scanners;
\getval{failed-share} of measurements failed.

values.md renders the same entries as a Markdown table for contexts without \getval (presentations, notes), values kept copy-pasteable and unescaped/unformatted for TeX.

Formatting (fmt) is applied only at write time — store[name] always returns the raw value, so derived computations never see formatted strings. Failing entries do not abort the run by default: they are logged, rendered as an ERR placeholder with the error as a % comment, and collect(strict=True) turns them into exceptions instead.

class ValueStore:
    """Insertion-ordered registry of named scalar values for a paper.

    Workflow: ``register()`` LazyFrame queries, eager DataFrames, or plain
    values; ``collect()`` to resolve them to scalars; optionally register
    derived plain values computed from ``store[name]``; emit via
    ``write()`` (``values.tex``) and/or ``write_markdown()`` (a Markdown
    table), whichever output strategy the context calls for.
    """

    def __init__(self) -> None:
        self._entries: dict[str, _Entry] = {}

    def register(
        self,
        name: str,
        value: Any,
        fmt: Optional[Formatter] = None,
        note: Optional[str] = None,
        emit: bool = True,
    ) -> "ValueStore":
        """Register a value under ``name``.

        Args:
            name: Key used in ``\\getval{name}``. Must not contain
                ``\\ { } % #`` or whitespace.
            value: A 1x1 ``pl.LazyFrame`` query (collected and extracted in
                ``collect()``), a 1x1 ``pl.DataFrame`` (extracted in
                ``collect()`` without collection), or any other value (stored
                verbatim, resolved immediately, no postprocessing).
            fmt: Python format spec string (e.g. ``",.1f"``) or callable
                ``value -> str``. Applied only in ``write()``.
            note: Comment written next to the entry in the output file, e.g.
                ``"FIXME hardcoded, see ..."``. Must not contain newlines.
            emit: When False the entry is resolved and readable via
                ``store[name]`` but never written to the output file.

        Returns:
            self, for chaining.
        """
        if not name or _NAME_FORBIDDEN.search(name):
            raise ValueError(
                f"invalid value name {name!r}: must be non-empty and must not "
                "contain '\\', '{', '}', '%', '#', or whitespace"
            )
        if name in self._entries:
            raise ValueError(f"duplicate value name {name!r}")
        if note is not None and ("\n" in note or "\r" in note):
            raise ValueError(
                f"note for value {name!r} contains a newline, which would "
                "terminate the TeX comment and leak the remainder as live code"
            )
        self._entries[name] = _Entry(value=value, fmt=fmt, note=note, emit=emit)
        return self

    def collect(
        self, individual: bool = False, strict: bool = False, **collect_kwargs: Any
    ) -> "ValueStore":
        """Resolve all pending LazyFrame/DataFrame entries to scalars.

        LazyFrame entries are collected — batched via ``pl.collect_all``
        (parallelism + common-subplan elimination) by default, or one at a
        time with ``individual=True`` (bounded memory). If the batched
        ``collect_all`` raises, all pending entries fall back to individual
        collection so one bad query doesn't take down the batch.

        Every resulting/registered DataFrame is then validated to be exactly
        1 row x 1 column and non-null, and replaced by ``df.item()``.

        Args:
            individual: Collect LazyFrames one at a time instead of batched.
            strict: Raise on the first failure instead of isolating it. With
                the default ``False``, a failing entry logs a warning and is
                stored as errored; ``write()`` renders it as an ``ERR``
                placeholder (if ``emit=True``).
            **collect_kwargs: Passed through to ``pl.collect_all`` /
                ``LazyFrame.collect``.

        Returns:
            self, for chaining. Idempotent: already resolved or errored
            entries are skipped (errored entries are not retried).
        """
        lazy = {
            name: entry
            for name, entry in self._entries.items()
            if entry.unresolved and isinstance(entry.value, pl.LazyFrame)
        }
        if lazy:
            if individual:
                self._collect_individual(lazy, strict, collect_kwargs)
            else:
                try:
                    frames = pl.collect_all(
                        [entry.value for entry in lazy.values()], **collect_kwargs
                    )
                    for entry, df in zip(lazy.values(), frames):
                        entry.value = df
                except Exception as exc:
                    logger.warning(
                        f"batched collect_all failed ({exc}); "
                        "falling back to individual collection"
                    )
                    self._collect_individual(lazy, strict, collect_kwargs)

        for name, entry in self._entries.items():
            if not (entry.unresolved and isinstance(entry.value, pl.DataFrame)):
                continue
            df = entry.value
            if df.height != 1 or df.width != 1:
                self._fail(
                    name,
                    entry,
                    f"value {name!r}: expected exactly 1 row x 1 column, "
                    f"got {df.height} rows x {df.width} columns",
                    strict,
                )
                continue
            item = df.item()
            if item is None:
                self._fail(name, entry, f"value {name!r} is null", strict)
                continue
            entry.value = item
        return self

    def _collect_individual(
        self, lazy: dict[str, _Entry], strict: bool, collect_kwargs: dict[str, Any]
    ) -> None:
        for name, entry in lazy.items():
            try:
                entry.value = entry.value.collect(**collect_kwargs)
            except Exception as exc:
                if strict:
                    raise
                self._fail(
                    name, entry, f"collecting value {name!r} failed: {exc}", False
                )

    @staticmethod
    def _fail(name: str, entry: _Entry, message: str, strict: bool) -> None:
        if strict:
            raise ValueError(message)
        entry.error = message
        logger.warning(message)

    def __getitem__(self, name: str) -> Any:
        """Raw resolved value, for post-collect derived computations."""
        if name not in self._entries:
            raise KeyError(f"unknown value {name!r}")
        entry = self._entries[name]
        if entry.errored:
            raise RuntimeError(
                f"value {name!r} errored during collect(): {entry.error}"
            )
        if entry.unresolved:
            raise RuntimeError(
                f"value {name!r} is not resolved yet; call collect() first"
            )
        return entry.value

    def __contains__(self, name: str) -> bool:
        return name in self._entries

    def write(self, path: Union[str, Path], xspace: bool = True) -> None:
        """Write all ``emit=True`` entries to ``path`` as ``\\setval`` lines.

        Args:
            path: Output file, typically ``values.tex``.
            xspace: Append ``\\xspace`` to every value (requires
                ``\\usepackage{xspace}`` in the paper).
        """
        self._require_resolved()
        lines = [_PREAMBLE]
        for name, entry in self._entries.items():
            if not entry.emit:
                continue
            if entry.errored:
                assert entry.error is not None
                comment = " ".join(entry.error.split())
                lines.append(f"\\setval{{{name}}}{{ERR}} % ERROR: {comment}")
                continue
            text = self._format(name, entry)
            value = _escape_tex(text)
            if xspace:
                value += "\\xspace"
            line = f"\\setval{{{name}}}{{{value}}}"
            if entry.note is not None:
                line += f" % {entry.note}"
            lines.append(line)
        Path(path).write_text("\n".join(lines) + "\n")

    def to_markdown(self) -> str:
        """Render all ``emit=True`` entries as a Markdown table.

        Values are formatted with the same ``fmt`` as in ``write()`` but
        without TeX escaping or ``\\xspace``, so they can be copied verbatim
        into presentations or notes. Errored entries render as ``ERR`` with
        the error message in the note column.

        Returns:
            The table as a string, ending in a newline.
        """
        self._require_resolved()
        rows = []
        for name, entry in self._entries.items():
            if not entry.emit:
                continue
            if entry.errored:
                assert entry.error is not None
                value = "ERR"
                note = f"ERROR: {' '.join(entry.error.split())}"
            else:
                value = self._format(name, entry)
                note = entry.note or ""
            rows.append((name, value, note))

        header = ("Name", "Value", "Note")
        if not any(note for _, _, note in rows):
            header = header[:2]
            rows = [row[:2] for row in rows]
        cells = [header, *(
            tuple(_escape_markdown(cell) for cell in row) for row in rows
        )]
        widths = [max(len(row[i]) for row in cells) for i in range(len(header))]
        lines = [
            "| " + " | ".join(cell.ljust(w) for cell, w in zip(row, widths)) + " |"
            for row in cells
        ]
        lines.insert(1, "|" + "|".join("-" * (w + 2) for w in widths) + "|")
        return "\n".join(lines) + "\n"

    def write_markdown(self, path: Union[str, Path]) -> None:
        """Write the ``to_markdown()`` table to ``path``, e.g. ``values.md``."""
        Path(path).write_text(self.to_markdown())

    def _require_resolved(self) -> None:
        unresolved = [
            name for name, entry in self._entries.items() if entry.unresolved
        ]
        if unresolved:
            raise RuntimeError(
                f"cannot write: unresolved entries {unresolved}; "
                "call collect() first"
            )

    @staticmethod
    def _format(name: str, entry: _Entry) -> str:
        if entry.fmt is None:
            text = str(entry.value)
        elif isinstance(entry.fmt, str):
            text = format(entry.value, entry.fmt)
        else:
            text = str(entry.fmt(entry.value))
        if "\n" in text or "\r" in text:
            raise ValueError(
                f"formatted value for value {name!r} contains a newline"
            )
        return text

            lines.append(f"\\setval{{{name}}}{{ERR}} % ERROR: {comment}")
            continue
        text = self._format(name, entry)
        value = _escape_tex(text)
        if xspace:
            value += "\\xspace"
        line = f"\\setval{{{name}}}{{{value}}}"
        if entry.note is not None:
            line += f" % {entry.note}"
        lines.append(line)
    Path(path).write_text("\n".join(lines) + "\n")

def fmt_percent(digits: int = 0, factor: float = 1) -> Callable[[Any], str]:
    """Percentage, e.g. ``13.0`` -> ``13%`` (escaped to ``13\\%`` by the writer).

    Args:
        digits: Number of decimal places.
        factor: Multiplier applied before formatting; use 100 if the value is
            a fraction (e.g. ``0.13``) rather than already in percent.
    """

    def fmt(value: Any) -> str:
        return f"{value * factor:.{digits}f}%"

    return fmt
